  function animatePipeline(totalMs) {

    const weights = [0.38, 0.10, 0.12, 0.10, 0.10, 0.10, 0.10];

    let elapsed = 0;

    weights.forEach((w, i) => {

        </div>`).join('')}`;

    // Load segment 0 immediately, others on demand

    for (let i = 1; i < segs.length; i++) {

        document.querySelector(`.stab:nth-child(${idx+1})`).addEventListener('click', function onceClick() {

  function switchSegTab(i) {

    document.querySelectorAll('.stab').forEach((t, j) => t.classList.toggle('active', i === j));

  }
